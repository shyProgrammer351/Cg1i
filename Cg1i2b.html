 printMap() { 
	( int row = 0; row < 7; row++ ) {																					 
		( int column = 0; column < 9; column++ ) {																				 
			( map[row][column] == 'W' ) 
				printf(" \033[44m   \033[0m");			 
			( map[row][column] == ' ' ) 
				printf("  "); 
			( map[row][column] == 'E' ) 
				printf(" \033[34m문 \033[0m");			 
		} 																			 
		printf("\n"); 
	}																					 
}
