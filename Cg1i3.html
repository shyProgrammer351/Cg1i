Fb	void raise( char *character, int action, int *hp, int *xp, int *tn ) {

		// 캐릭터 활동별 능력치 초기화

		int stat[4][3] = { { 2, 1, 2 },

		                   { 1, 3, 4 },

		                   { 4, 4, 3 },

		                   { 3, 2, 1 } };

	 

		char image[4][64] = { "  _\n_| |_\n(-_-)            \n(   )            \n",

		                      "  _\n_| |_\n(^O^) ~♩~~♪~♬~♩~\n(   )            \n",

		                      "  _\n_| |_\n(O.O)  O   O     \n(   ) \\|/ \\|/    \n",

		                      "  _\n_| |_\n(@O@)            \n(   ) |_|>       \n" };

	 

		// 캐릭터 능력치 업데이트

		*hp += stat[action-1][0];

		*xp += stat[action-1][1];

		*tn += stat[action-1][2];

		strcpy( character, image[action-1] );

Fe	}

	 

	// 프로그램시작
